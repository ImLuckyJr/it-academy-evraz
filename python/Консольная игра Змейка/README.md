## Консольная игра "Змейка" на Python

```python
import curses  # Импортируем модуль curses
from curses import KEY_RIGHT, KEY_LEFT, KEY_UP, KEY_DOWN  # Из модуля curses импортируем события нажатия на стрелки (вверх, вниз, вправо, влево)
from random import randint  # Из модуля random импортируем функцию генерации случайного целого числа

curses.initscr()  # Инициализируем экран игры
window = curses.newwin(30, 60, 0, 0)  # Создаем окно высотой 30px и шириной 60px, отсутп по y = 0, отступ по x = 0

key = KEY_RIGHT  # Переменная, в которой будет храниться нажатая клавиша (по умолчанию: стрелка вправо)
score = 0  # Количество очков, которые заработал игрок

snake = [[5, 8], [5, 7], [5, 6]]  # Изначальные координаты змейки [[y1, x1], [y2, x2], [y3, x3]]
food = [randint(1, 29), randint(1, 59)]  # Изначальные координаты еды

# window.addch(y, x, 'символ') - добавление символа на экран
window.addch(food[0], food[1], '#')  # Отрисовка еды в игровом окне

# Игра запущена, пока не будет нажата клавиша Esc (номер клавиши: 27)
while key != 27:
    window.border(0)  # Добавление рамки для окна (игровая область)
    # window.addStr(y, x, 'строка') - добавление строки на экран
    window.addstr(0, 2, 'Количество очков: ' + str(score) + ' ')  # Отрисовка количества набранных очков
    # window.timeout() - задержка между сменой кадров экрана
    window.timeout(140 - (len(snake) / 5 + len(snake) / 10) % 120)  # Уменьшаем задержку по мере увеличения набранных очков

    event = window.getch()  # Ожидаем нажатия клавиши от игрока

    # Если была нажата какая-то клавиша (клавиша не равна -1), то переназначаем переменную с нажатой клавишей
    if event != -1:
        key = event

    # Считаем новые координаты головы змейки
    # Если нажата стрелка вниз, то увеличиваем координату y на 1
    if (key == KEY_DOWN):
        snake[0][0] += 1 

    # Если нажата стрелка вверх, то уменьшаем координату y на 1 
    if (key == KEY_UP):
        snake[0][0] -= 1 

    # Если нажата стрелка влево, то уменьшаем координату x на 1 
    if (key == KEY_LEFT):
        snake[0][1] -= 1 
        
    # Если нажата стрелка вправо, то увеличиваем координату x на 1
    if (key == KEY_RIGHT):
        snake[0][1] += 1 
    
    # Добавляем новую координату головы змейки в список с координатами змейки   
    snake.insert(0, [snake[0][0], snake[0][1]]) 

    # Завершаем игру, когда змейка врезалась в стенку (координата головы змейки равна какой-либо из координат границы окна)
    if snake[0][0] == 0 or snake[0][0] == 29 or snake[0][1] == 0 or snake[0][1] == 59:
        break

    # Завершаем игру, когда змейка врезалась в себя (координата головы змейки равна какой-либо из остальных координат змейки)
    if snake[0] in snake[1:]:
        break

    # Ситуация, когда змейка съедает еду
    # Если координата головы змейки равна еде, значит, змейка съела еду
    # Иначе просто отрезаем хвост змейки и добавляем его в начало (имитируем передвижение змейки)
    if snake[0] == food:
        score += 1  # Увеличиваем количество набранных очков на единицу
        food = [randint(1, 28), randint(1, 58)]  # Задаем новые случайные координаты для еды
        window.addch(food[0], food[1], '#')  # Заново отрисовываем еду с новыми координатами
    else:
        last = snake.pop()  # Отрезаем хвост змейки (последнюю координату)
        window.addch(last[0], last[1], ' ')  # Стриаем хвост в игровом окне (заменяем его на пробел)
    window.addch(snake[0][0], snake[0][1], '*')  # Отрисовываем добавленную ранее координату головы змейки

curses.endwin()  # Закрываем игровое окно
print("\Количество набранных очков: " + str(score))  # Показываем набранное количество очков игроку
```
